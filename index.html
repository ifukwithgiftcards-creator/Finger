let index = 0; // Spawn words every ~600ms setInterval(() => { spawnWord(words[index]); index = (index + 1) % words.length; }, 600); function spawnWord(text) { const el = document.createElement("div"); el.className = "word"; el.innerText = text; // Random trajectory const tx = (Math.random() - 0.5) * 2000; const ty = (Math.random() - 0.5) * 1200; const rot = (Math.random() - 0.5) * 720; stage.appendChild(el); // Start far away el.style.transform = ` translate(-50%, -50%) translate3d(0, 0, -2500px) scale(0.2) rotate(0deg) `; // Force layout el.offsetHeight; // Animate forward el.style.transition = "transform 2.2s cubic-bezier(0.1, 0.8, 0.2, 1), opacity 2.2s"; el.style.transform = ` translate(-50%, -50%) translate3d(${tx}px, ${ty}px, 1500px) scale(3) rotate(${rot}deg) `; el.style.opacity = 0; // Cleanup setTimeout(() => { el.remove(); }, 2200); } // Starfield loop (~50ms) setInterval(() => { spawnStar(); }, 50); function spawnStar() { const star = document.createElement("div"); star.className = "star"; const x = Math.random() * window.innerWidth; const y = Math.random() * window.innerHeight; stage.appendChild(star); star.style.left = x + "px"; star.style.top = y + "px"; // Start deep star.style.transform = `translateZ(-2000px) scale(0.5)`; star.offsetHeight; // Animate forward star.style.transition = "transform 1.5s linear, opacity 1.5s"; star.style.opacity = 1; star.style.transform = `translateZ(1000px) scale(2)`; setTimeout(() => { star.style.opacity = 0; }, 1200); setTimeout(() => { star.remove(); }, 1500); }
